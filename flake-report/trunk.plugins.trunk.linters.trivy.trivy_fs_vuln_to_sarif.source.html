7 # OSV/GHSA: https://docs.github.com/en/code-security/security-advisories/global-security-advisories/about-the-github-advisory-database#about-cvss-levels
8 # SARIF: https://docs.oasis-open.org/sarif/sarif/v2.0/csprd02/sarif-v2.0-csprd02.html#_Toc10127839
9 SARIF_SEVERITY_BY_OSV_SEVERITY = {
10 "CRITICAL": "error",
11 "HIGH": "error",
12 "MODERATE": "warning",
13 "MEDIUM": "warning",
14 "LOW": "note",
15 }
16  
17 DEFAULT_SARIF_SEVERITY = "error"
18  
19  
20 def get_sarif_severity(vuln) -> str:
21 """Get the SARIF severity appropriate for a given OSV vulnerability entry."""
22 if "Severity" not in vuln:
23 return DEFAULT_SARIF_SEVERITY
24  
25 severity = vuln["Severity"].upper()
26  
27 return SARIF_SEVERITY_BY_OSV_SEVERITY.get(severity, DEFAULT_SARIF_SEVERITY)

30 def to_result_sarif(
31 path: str, severity: str, vuln_id: str, description: str, lineno: int
32 ):
33 return {
34 "level": severity,
35 "locations": [
36 {
37 "physicalLocation": {
38 "artifactLocation": {
39 "uri": path,
40 },
41 "region": {
42 "startColumn": 0,
43 "startLine": lineno,
44 },
45 }
46 }
47 ],
48 "message": {
49 "text": description,
50 },
51 "ruleId": vuln_id,
52 }

55 def main(argv):
56 trivy_json = json.load(sys.stdin)

60 for result in trivy_json.get("Results", []):

69 if path not in lockfiles:

72 if description[-1] != ".":

74

78
